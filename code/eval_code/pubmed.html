Fetch

Extract counts

Create function.

fetch_counts <- function(search_term) {
  
  search_term %>% 
    paste0("(", ., ")") %>%
    # paste0("(", ., "[Title/Abstract])") %>%  # to only search title/abstract
    paste(collapse = " OR ") %>% 
    rentrez::entrez_search(
      db = "pubmed",
      term = .,
      use_history = F
    ) %>% 
    magrittr::extract2("count")
  
}

Run.

# Tidy names
tidy_names <- c(
  `ClinicalTrials.gov` = "clinicaltrials",
  `PROSPERO` = "prospero",
  `ISRCTN` = "isrctn",
  `ANZCTR` = "anzctr",
  `ChiCTR` = "chictr",
  `JPRN` = "jprn"
)

# Run
pubmed <- 
  map(search_terms, fetch_counts) %>% 
  as_tibble() %>% 
  rename(tidy_names) %>% 
  gather(registers, counts)

# Save
write_csv(pubmed, "../../data/interim_data/pubmed.csv")

Extract all data

esearch <- rentrez::entrez_search(
    db = "pubmed",
    term = search_term,
    use_history = T
  )

retstart <- 
  esearch %>% 
  magrittr::extract2("count") %>% 
  floor() %>% 
  seq(0, ., 1) %>% 
  multiply_by(10000)

pubmed <- 
  retstart %>% 
  map(~ rentrez::entrez_fetch(
    db = "pubmed",
    web_history = esearch$web_history,
    retstart = .x,  # not inclusive
    retmax = 10000,
    rettype = "xml"
  )) %>% 
  unlist() %>% 
  xml2::read_xml() %>% 
  metareadr::mt_extract_pubmed()

write_csv(pubmed, "../../data/interim_data/pubmed.csv")

Plot

pubmed %>% 
  mutate(
    registers = fct_reorder(registers, -counts),
    n = scales::comma(counts),
    p = scales::percent(counts / sum(counts)),
    n_p = glue::glue("{n} ({p})")
  ) %>% 
  ggplot(aes(x = registers, y = counts, fill = registers, color = registers)) +
  geom_col(alpha = 0.75, show.legend = F) +
  geom_text(
    aes(label = n_p), 
    size = 3, 
    vjust = -0.5, 
    show.legend = F
  ) +
  labs(x = NULL, y = "Number of titles or abstracts") +
  scale_y_continuous(labels = scales::comma) +
  theme(
    panel.grid.major.x = element_blank(),
    axis.text.x = element_text(margin = margin(t = -10))
  )

# Save
ggsave("../../output/figure_output/pubmed.jpg", width = 8, height = 6)
